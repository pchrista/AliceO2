29 class Triggers
30 {
31  public:
32  enum { bitA = 0,
33  bitC = 1, // alias of bitAIn (FT0/FDD)
34  bitAIn = 1, // alias of bitC (FV0)
35  bitVertex = 2, // alias of bitAOut (FT0/FDD)
36  bitAOut = 2, // alias of bitVertex (FV0)
37  bitCen = 3,
38  bitSCen = 4,
39  bitLaser = 5, // indicates the laser was triggered in this BC
40  bitOutputsAreBlocked = 6, // indicates that laser-induced pulses should arrive from detector to FEE in this BC (and trigger outputs are blocked)
41  bitDataIsValid = 7 };
42  static const int16_t DEFAULT_TIME = -5000; // for average of one side (A or C)
43  static const int16_t DEFAULT_AMP = 0;
44  static const int16_t DEFAULT_ZERO = 0;

46  Triggers() = default;
47  Triggers(uint8_t signals, int8_t chanA, int8_t chanC, int32_t aamplA, int32_t aamplC, int16_t atimeA, int16_t atimeC)
48  {
49  triggersignals = signals;
50  nChanA = chanA;
51  nChanC = chanC;
52  amplA = aamplA;
53  amplC = aamplC;
54  timeA = atimeA;
55  timeC = atimeC;
56  }
57  bool getOrA() const { return (triggersignals & (1 << bitA)) != 0; }
58  bool getOrC() const { return (triggersignals & (1 << bitC)) != 0; } // only used by FT0/FDD (same bit as OrAIn in FV0)
59  bool getOrAIn() const { return (triggersignals & (1 << bitAIn)) != 0; } // only used by FV0 (same bit as OrC in FT0/FDD)
60  bool getVertex() const { return (triggersignals & (1 << bitVertex)) != 0; } // only used by FT0/FDD (same bit as OrAOut in FV0
61  bool getOrAOut() const { return (triggersignals & (1 << bitAOut)) != 0; } // only used by FV0 (same bit as Vertex in FT0/FDD)
62  bool getCen() const { return (triggersignals & (1 << bitCen)) != 0; }
63  bool getSCen() const { return (triggersignals & (1 << bitSCen)) != 0; }
64  bool getLaser() const { return (triggersignals & (1 << bitLaser)) != 0; }
65  bool getLaserBit() const { return getLaser(); } // TODO: remove after QC is modified
66  bool getOutputsAreBlocked() const { return (triggersignals & (1 << bitOutputsAreBlocked)) != 0; }
67  bool getDataIsValid() const { return (triggersignals & (1 << bitDataIsValid)) != 0; }

88  void setTriggers(Bool_t isA, Bool_t isC, Bool_t isVrtx, Bool_t isCnt, Bool_t isSCnt, int8_t chanA, int8_t chanC, int32_t aamplA,
89  int32_t aamplC, int16_t atimeA, int16_t atimeC, Bool_t isLaser, Bool_t isOutputsAreBlocked, Bool_t isDataValid)
90  {
91  uint8_t trgsig = (isA << bitA) | (isC << bitC) | (isVrtx << bitVertex) | (isCnt << bitCen) | (isSCnt << bitSCen) | (isLaser << bitLaser) | (isOutputsAreBlocked << bitOutputsAreBlocked) | (isDataValid << bitDataIsValid);
92  setTriggers(trgsig, chanA, chanC, aamplA, aamplC, atimeA, atimeC);
93  }

95  void cleanTriggers()

107  }
