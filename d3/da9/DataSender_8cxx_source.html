79 void DataSender::send(FairMQParts& parts, ChannelIndex channelIndex)
80 {
81  mPolicy.send(mProxy, parts, channelIndex);
82 
88  TimesliceIndex& index = mRegistry.get<TimesliceIndex>();
89  index.updateOldestPossibleOutput();
90 
91  auto oldest = index.getOldestPossibleOutput();
92  if (oldest.timeslice.value == -1) {
93  return;
94  }
95  auto* channel = mProxy.getOutputChannel(channelIndex);
96 
97  FairMQParts oldestParts;
98  FairMQMessagePtr payload(channel->Transport()->CreateMessage());
99  DomainInfoHeader dih;
100  dih.oldestPossibleTimeslice = oldest.timeslice.value;
101  auto channelAlloc = o2::pmr::getTransportAllocator(channel->Transport());
102  auto header = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dih});
103  // sigh... See if we can avoid having it const by not
104  // exposing it to the user in the first place.
105  oldestParts.AddPart(std::move(header));
106  oldestParts.AddPart(std::move(payload));
107  LOGP(debug, "Notifying {} {} about oldest possible timeslice being {} from {} #{}",
108  channelIndex.value,
109  channel->GetName(),
110  oldest.timeslice.value,
111  oldest.channel.value == -1 ? "slot" : " input channel",
112  oldest.channel.value == -1 ? oldest.slot.index : oldest.channel.value);
113  channel->Send(oldestParts);
114 }
