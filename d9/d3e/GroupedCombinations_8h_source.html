149  std::tuple<As...> getAssociatedTables()
150  {
151  auto& currentGrouping = GroupingPolicy::mCurrent;
152  constexpr auto k = sizeof...(As);
153  auto slicerIterators = functionToTuple<k>(&GroupSlicer<G, Us...>::begin, *mSlicer);
154  o2::soa::for_<k>([&](auto i) {
155  auto col = std::get<i.value>(currentGrouping);
156  for (auto& slice : *mSlicer) {
157  if (slice.groupingElement().globalIndex() == col.globalIndex()) {
158  std::get<i.value>(slicerIterators) = slice;
159  break;
160  }
161  }
162  });
163 
164  return getSlices(slicerIterators, std::index_sequence_for<As...>());
165  }

168  auto getSliceAt(std::tuple<Ts...>& t)
169  {

171  auto associatedType = it.template prepareArgument<T>();

173  }

181  void setCurrentGroupedCombination()
182  {
183  std::tuple<As...> initAssociatedTables = getAssociatedTables();
184  constexpr auto k = sizeof...(As);
185  bool moveForward = false;
186  o2::soa::for_<k>([&](auto i) {
187  if (std::get<i.value>(initAssociatedTables).size() == 0) {
188  moveForward = true;
189  }
190  });
191  while (!this->mIsEnd && moveForward) {
192  GroupingPolicy::addOne();
193  std::tuple<As...> temp = getAssociatedTables();
194  moveForward = false;
195  o2::soa::for_<k>([&](auto i) {
196  if (std::get<i.value>(temp).size() == 0) {
197  moveForward = true;
198  }
199  });
200  }
201  std::tuple<As...> associatedTables = getAssociatedTables();
202 
203  if (!this->mIsEnd) {
204  auto& currentGrouping = GroupingPolicy::mCurrent;
205  o2::soa::for_<k>([&](auto i) {
206  std::get<i.value>(associatedTables).bindExternalIndices(mGrouping.get());
207  });
208 
209  mCurrentGrouped.emplace(interleaveTuples(currentGrouping, associatedTables));
210  }
211  }

213  std::shared_ptr<GroupSlicer<G, Us...>> mSlicer = nullptr;

215  std::optional<GroupedIteratorType> mCurrentGrouped;
216  };
