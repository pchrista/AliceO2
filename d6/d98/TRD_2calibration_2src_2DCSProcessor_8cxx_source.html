30 void TRDDCSMinMaxMeanInfo::addPoint(float value)
31 {
32  if (nPoints == 0) {
33  minValue = value;
34  maxValue = value;
35  meanValue = value;
36  } else {
37  if (value < minValue) {
38  minValue = value;
39  }
40  if (value > maxValue) {
41  maxValue = value;
42  }
43  meanValue += (value - meanValue) / (nPoints + 1);
44  }
45  ++nPoints;
46 }

62 int DCSProcessor::process(const gsl::span<const DPCOM> dps)
63 {
64 
65  // first we check which DPs are missing - if some are, it means that
66  // the delta map was sent
67  if (mVerbose) {
68  LOG(info) << "\n\n\nProcessing new TF\n-----------------";
69  }
70  if (!mStartTFset) {
71  mStartTF = mStartValidity;
72  mStartTFset = true;
73  }
74 
75  std::unordered_map<DPID, DPVAL> mapin;
76  for (auto& it : dps) {
77  mapin[it.id] = it.data;
78  }
79  for (auto& it : mPids) {
80  const auto& el = mapin.find(it.first);
81  if (mVerbose) {
82  if (el == mapin.end()) {
83  LOG(info) << "DP " << it.first << " not found in map";
84  } else {
85  LOG(info) << "DP " << it.first << " found in map";
86  }
87  }
88  }
89 
90  // now we process all DPs, one by one
91  for (const auto& it : dps) {
92  // we process only the DPs defined in the configuration
93  const auto& el = mPids.find(it.id);
94  if (el == mPids.end()) {
95  if (mVerbose) {
96  LOG(info) << "DP " << it.id << " not found in DCSProcessor, we will not process it";
97  }
98  continue;
99  }
100  processDP(it);
101  mPids[it.id] = true;
102  }
103 
104  return 0;
105 }

109 int DCSProcessor::processDP(const DPCOM& dpcom)
110 {
111 
112  // processing single DP
113 
114  auto& dpid = dpcom.id;
115  const auto& type = dpid.get_type();
116  if (mVerbose) {
117  if (type == DPVAL_DOUBLE) {
118  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<double>(dpcom);
119  } else if (type == DPVAL_INT) {
120  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<int32_t>(dpcom);
121  }
122  }
123  auto flags = dpcom.data.get_flags();
124  if (processFlags(flags, dpid.get_alias()) == 0) {
125  if (type == DPVAL_DOUBLE) {
126  auto& dvect = mDpsDoublesmap[dpid];
127  if (mVerbose) {
128  LOG(info) << "mDpsDoublesmap[dpid].size() = " << dvect.size();
129  }
130  auto etime = dpcom.data.get_epoch_time();
131  if (dvect.size() == 0 || etime != dvect.back().data.get_epoch_time()) {
132  // only add data point in case it was not already read before
133  dvect.push_back(dpcom);
134  }
135  }
136  if (type == DPVAL_INT) {
137  // TODO so far there is no processing at all for these type of DCS data points
138  if (std::strstr(dpid.get_alias(), "trd_runNo") != nullptr) { // DP is trd_runNo
139  std::string aliasStr(dpid.get_alias());
140  auto runNumber = o2::dcs::getValue<int32_t>(dpcom);
141  if (mVerbose) {
142  LOG(info) << "Run number = " << runNumber;
143  }
144  // end processing current DP, when it is of type trd_runNo
145  } else if (std::strstr(dpid.get_alias(), "trd_runType") != nullptr) { // DP is trd_runType
146  std::string aliasStr(dpid.get_alias());
147  auto runType = o2::dcs::getValue<int32_t>(dpcom);
148  if (mVerbose) {
149  LOG(info) << "Run type = " << runType;
150  }
151  }
152  }
153  }
154  return 0;
155 }
