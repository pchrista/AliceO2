133 int CTPConfiguration::processConfigurationLineRun3(std::string& line, int& level)
134 {
135  // LOG(debug) << "line:" << line;
136  if (line.size() == 0) {
137  return 0;
138  }
139  if (line.at(0) == '#') {
140  return 0;
141  }
142  //
143  std::vector<std::string> tokens = o2::utils::Str::tokenize(line, ' ');
144  size_t ntokens = tokens.size();
145  if (ntokens == 0) {
146  LOG(warning) << "# of tokens zero in line:" << line;
147  return 0;
148  }
149  size_t first;
150  if (CTPGenerator::Generators.count(tokens[0])) {
151  if (level != CLASS) {
152  level = GENS;
153  }
154  level = GENS;
155  } else if ((first = line.find("bcm")) != std::string::npos) {
156  if (level == MASKS) {
157  level = MASKS;
158  }
159  } else if ((first = line.find("LTG")) != std::string::npos) {
160  level = LTG;
161  } else if ((first = line.find("cluster")) != std::string::npos) {
162  level = CLUSTER;
163  } else {
164  }
165  switch (level) {
166  case MASKS: {
167  BCMask bcmask;
168  bcmask.name = tokens[1];
169  std::regex LHdelimiters("[LH]");
170  if (std::regex_search(tokens[2], LHdelimiters)) {
171  // jusko notation
172  } else {
173  // list of integers
174  for (int i = 2; i < ntokens; i++) {
175  uint32_t bc;
176  try {
177  bc = std::stoull(tokens[i]);
178  } catch (...) {
179  LOG(info) << "mask syntax:" << tokens[i];
180  continue;
181  }
182  bcmask.BCmask.set(bc, 1);
183  }
184  }
185  mBCMasks.push_back(bcmask);
186  LOG(info) << "BC mask added:" << bcmask.name;
187  break;
188  }
189  case GENS: {
190  CTPGenerator gen;
191  gen.name = tokens[0];
192  gen.frequency = tokens[1];
193  LOG(info) << "Gen added:" << line;
194  break;
195  }
196  case LTG: {
197  CTPDetector ctpdet;
198  std::string detname = tokens[1];
199  capitaliseString(detname);
200  o2::detectors::DetID det(detname.c_str());
201  if (isDetector(det)) {
202  ctpdet.detID = det.getID();
203  LOG(info) << "Detector found:" << det.getID() << " " << detname;
204  } else {
205  LOG(info) << "Unknown detectors:" << line;
206  }
207  mDetectors.push_back(ctpdet);
208  level = LTGitems;
209  break;
210  }
211  case LTGitems: {
212  if (ntokens == 1) {
213  mDetectors.back().mode = tokens[0];
214  }
215  LOG(info) << "LTGitem:" << line;
216  break;
217  }
218  case CLUSTER: {
219  CTPCluster cluster;
220  try {
221  cluster.hwMask = std::stoull(tokens[0]);
222  } catch (...) {
223  LOG(info) << "Cluster syntax error:" << line;
224  return level;
225  }
226  LOG(info) << "Cluster:" << line;
227  cluster.name = tokens[2];
228  o2::detectors::DetID::mask_t mask;
229  for (int item = 3; item < ntokens; item++) {
230  std::string detname = tokens[item];
231  capitaliseString(detname);
232  // LOG(info) << "Detector:" << detname;
233  o2::detectors::DetID det(detname.c_str());
234  isDetector(det);
235  mask |= det.getMask();
236  }
237  cluster.maskCluster = mask;
238  mClusters.push_back(cluster);
239  level = CLASS;
240  // LOG(info) << "Cluster done:" << cluster.name << std::endl;
241  break;
242  }
243  case CLASS: {
244  // add to the last cluster
245  uint64_t index;
246  try {
247  index = std::stoull(tokens[0]);
248  } catch (...) {
249  LOG(info) << "Class syntax error:" << line;
250  return level;
251  }
252  LOG(info) << "Class:" << line;
253  CTPClass cls;
254  cls.classMask = 1ull << index;
255  cls.name = tokens[1];
256  cls.clusterIndex = mClusters.size() - 1;
257  // LOG(info) << "point:" << cls.cluster << " " << &mClusters.front();
258  mCTPClasses.push_back(cls);
259  break;
260  }
261  default: {
262  LOG(info) << "unknown line:" << line;
263  }
264  }
265  for (auto& cls : mCTPClasses) {
266  cls.cluster = &mClusters[cls.clusterIndex];
267  }
268  return 0;

270 int CTPConfiguration::processConfigurationLine(std::string& line, int& level)
271 {
272  if (line.size() == 0) {
273  return 0;
274  }
275  if (line.at(0) == '#') {
276  return 0;
277  }
278  size_t first;
279  if ((first = line.find("PARTITION:")) != std::string::npos) {
280  mName = o2::utils::Str::trim_copy(line.erase(first, 10));
281  return 0;
282  }
283  if ((first = line.find("VERSION:")) != std::string::npos) {
284  mVersion = o2::utils::Str::trim_copy(line.erase(first, 8));
285  return 0;
286  }
287  if ((first = line.find("INPUTS:")) != std::string::npos) {
288  level = 1;
289  return 0;
290  }
291  if ((first = line.find("DESCRIPTORS:")) != std::string::npos) {
292  level = 3;
293  return 0;
294  }
295  if ((first = line.find("CLUSTERS:")) != std::string::npos) {
296  level = 4;
297  return 0;
298  }
299  if ((first = line.find("CLASSES:")) != std::string::npos) {
300  level = 7;
301  return 0;
302  }
304  std::vector<std::string> tokens = o2::utils::Str::tokenize(line, ' ');
305  size_t ntokens = tokens.size();
306  if (ntokens == 0) {
307  return 0;
308  }
309  switch (level) {
310  case 1:
312  {
313  if (ntokens != 4) {
314  LOG(fatal) << "INPUTS syntax error in, wrong number of items, expected 4:" << line;
315  return level;
316  }
317  CTPInput inp;
318  inp.name = tokens[0];
319  std::string detName = tokens[1];
320  inp.level = tokens[2];
321  o2::detectors::DetID det(detName.c_str());
322  isDetector(det);
323  inp.detID = det.getID();
324  //std::cout << "id:" << det.getID() << " ndets:" << det.getNDetectors() << std::endl;
325  try {
326  inp.inputMask = std::stoull(tokens[3], nullptr, 0);
327  } catch (...) {
328  LOG(fatal) << "INPUTS syntax error in mask:" << line;
329  return level;
330  }
331  mInputs.push_back(inp);
332  break;
333  }
334  case 3:
336  {
337  CTPDescriptor desc;
338  desc.name = tokens[0];
339  tokens.erase(tokens.begin());
340  for (auto& item : tokens) {
341  //CTPInput *inp = const_cast<CTPInput*> (isInputInConfig(item));
342  CTPInput* inp = isInputInConfig(item);
343  if (inp == nullptr) {
344  LOG(fatal) << "DESCRIPTOR: input not in INPUTS:" << item << " LINE:" << line;
345  } else {
346  desc.inputs.push_back(inp);
347  }
348  }
349  // Create inputs and mask
350  mDescriptors.push_back(desc);
351  break;
352  }
353  case 4:
355  {
356  CTPCluster cluster;
357  cluster.name = tokens[0];
358  tokens.erase(tokens.begin());
359  o2::detectors::DetID::mask_t mask;
360  for (auto& item : tokens) {
361  o2::detectors::DetID det(item.c_str());
362  isDetector(det);
363  mask |= det.getMask();
364  }
365  cluster.maskCluster = mask;
366  mClusters.push_back(cluster);
367  break;
368  }
369  case 7:
371  {
372  CTPClass cls;
373  if (ntokens != 4) {
374  LOG(fatal) << "CLASSES syntax error, wrong number of items, expected 4:" << line;
375  return level;
376  }
377  cls.name = tokens[0];
378  try {
379  cls.classMask = std::stoull(tokens[1]);
380  } catch (...) {
381  LOG(fatal) << "CLASSES syntax error in mask:" << line;
382  return level;
383  }
384  std::string token = tokens[2];
385  auto it = std::find_if(mDescriptors.begin(), mDescriptors.end(), [&token](const CTPDescriptor& obj) { return obj.name == token; });
386  if (it != mDescriptors.end()) {
387  cls.descriptor = &*it;
388  } else {
390  LOG(fatal) << "CLASSES syntax error, descriptor not found:" << token;
391  }
393  token = tokens[3];
394  auto it2 = std::find_if(mClusters.begin(), mClusters.end(), [&token](const CTPCluster& obj) { return obj.name == token; });
395  if (it2 != mClusters.end()) {
396  cls.cluster = &*it2;
397  } else {
399  LOG(fatal) << "CLASSES syntax error, cluster not found:" << token;
400  }
401  mCTPClasses.push_back(cls);
402  break;
403  }
404  default: {
405  LOG(fatal) << "CTP Config parser Unknown level:" << level;
406  }
407  }
408  return 0;
409 }
